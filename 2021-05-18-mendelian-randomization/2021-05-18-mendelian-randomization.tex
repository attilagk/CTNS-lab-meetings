\begin{frame}{Mendelian randomization}
\begin{columns}[t]
\begin{column}{0.4\textwidth}
What is $Z$?
\small
\begin{itemize}
\item instrument for $X$
\item a genetic variant
\end{itemize}

\includegraphics[scale=1.0]{figures/by-me/mendelian-randomization/mendelian-randomization.pdf}
\end{column}

\begin{column}{0.6\textwidth}
Linear models
	\small
\begin{eqnarray*}
	\mathrm{E}[Y|X, Z] &=& \beta_{XY} \mathrm{E}[X|Z] + \alpha_{XY} \\
	\mathrm{E}[X|Z] &=& \beta_{ZX} Z + \alpha_{ZX} \\
	\beta_{ZY} &=& \beta_{ZX} \beta_{XY}
\end{eqnarray*}
